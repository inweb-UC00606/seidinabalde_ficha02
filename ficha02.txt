#include <iostream>

int main() {

    std::cout << "Menu:" << std::endl;
    std::cout << "1 - Calcular a medida em polegadas:" << std::endl;
    std::cout << "2 - Calcular a temperatura em graus celsius:" << std::endl;

    int opcao;
    std::cin >> opcao;

    if (opcao == 1) {
        float polegadas, centimetros;

        std::cout << "Introduza a medida em polegadas: ";
        std::cin >> polegadas;

        centimetros = polegadas * 2.54;

        std::cout << polegadas << " polegadas equivalem a "
             << centimetros << " centimetros." << std::endl;

    } else if (opcao == 2) {
        float celsius;

        std::cout << "Introduza a temperatura em graus celsius: ";
        std::cin >> celsius;

        float far = celsius * 1.8 + 32;
        float kel = celsius + 273.15;

        std::cout << "Temperatura em Fahrenheit: " << far << std::endl;
        std::cout << "Temperatura em Kelvin: " << kel << std::endl;

    } else {
        std::cout << "Opção inválida" << std::endl;
    }

    return 0;
}
