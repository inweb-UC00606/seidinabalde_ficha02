int main(){
    
    std::cout << "Menu:" << std::endl;
    std::cout << "1 - Calcular a medida em polegadas:" << std::endl;
    std::cout << "2 - Calcular a temperatura em graus celsius:" << std::endl;
    
    int opcao;
    
    std::cin >> opcao;
    
       if (opcao == 1){
        //declaracao das variáveis

    	//output
    	std::cout << "Introduza a medida em polegadas: ";

    	std::cin >> polegadas;
    
    	//processamento do calculo
    	centimetros = polegadas * 2.54;

       }else if(opcao == 2){

    	std::cout << "Temperatura em Kelvin: " << kel << endl;
        
    }else{
